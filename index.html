function showCard(index) { const c = cards[index]; if (!c) return; // UI Elementen ophalen const storeNameElem = document.getElementById('store-name'); const bc = document.getElementById('barcode'); const qr = document.getElementById('qrcode'); const txt = document.getElementById('display-text'); if (storeNameElem) storeNameElem.innerText = c.name || ''; // Reset weergave: alles verbergen if (bc) bc.style.display = 'none'; if (qr) { qr.style.display = 'none'; qr.innerHTML = ''; } if (txt) txt.innerText = c.displayNum || c.code || ''; // Data schoonmaken const cleanCode = String(c.code).trim(); if (c.type === 'qrcode') { if (qr) { qr.style.display = 'block'; try { new QRCode(qr, { text: cleanCode, width: 220, height: 220, correctLevel: QRCode.CorrectLevel.M }); } catch(e) { qr.innerText = "QR Fout: Data te lang"; } } } else { // BARCODE LOGICA (Verbeterd) if (bc) { bc.style.display = 'block'; try { // We gebruiken CODE128: dit accepteert letters, cijfers en tekens JsBarcode("#barcode", cleanCode, { format: "CODE128", width: 2, height: 100, displayValue: false, // We tonen het nummer zelf al in 'display-text' margin: 10 }); } catch(e) { console.error("Barcode tekenfout", e); bc.style.display = 'none'; if (qr) { qr.style.display = 'block'; qr.innerHTML = `
Geen barcode mogelijk voor deze code
`; } } } } window.scrollTo({top: 0, behavior: 'smooth'}); }
